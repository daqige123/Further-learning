# 实现一个函数 asyncPool(limit, tasks, callback)，限制并发数为 limit，全部完成后执行 callback

async function asyncPool(limit, tasks, callback) {
    let curExe = [];
    for (let i= 0; i < tasks.length; i++) {
        let curTask = Promise.rosolve(tasks[i])();
        curExe.push(curTask);
        curTask().finnally(()=> {
            let index =  curTask.indexof(curTask);
            curExe.splice(index, 1,);
        })
        if (curExe.length >= limit) {
            await Promise.race(curExe);
        }
    }
    return Promise.all(curExe).then(callback);
}
